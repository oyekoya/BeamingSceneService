00546 void SceneInterface::updateDirections(QStringList ids){
00547     if(!isConnectedToScene()){ // only allow updates if the scene is not running
00548         CMatrix mVisitor; // Transformation matrix class used in loop
00549         QReadLocker srlocker(&infoMutex);
00550         myquat.ToMatrix(mVisitor.mf); // fill out a transformation matrix with this node
00551         mVisitor=mVisitor.InvertRot(); // apparently they cannot agree with themselves on the basis of rotation/translation...
00552         mVisitor.mf[12] = mytranslation.x;
00553         mVisitor.mf[13] = mytranslation.y;
00554         mVisitor.mf[14] = mytranslation.z;
00555         mVisitor=mVisitor.InvertSimple(); // invert the visitor transformation so we now have a description of origo from visitor point of view
00556         srlocker.unlock();
00557         QWriteLocker swlocker(&infoMutex); // lock the scene info for writing
00558         for(int id=0;id<ids.size();id++){ // for each other client
00559             CVec3 finaltrans = mVisitor*sceneData[ids[id]].translation;
00560 #ifdef SCENEDBG
00561             qDebug() << "Final translation:" << finaltrans.x << " " << finaltrans.y << " " << finaltrans.z << " Distance: " << finaltrans.Distance(CVec3());
00562 #endif // SCENEDBG
00563             // The Beaming Scene reference should be right handed cartesian coordinate system:
00564             // x forward, y up, z right
00565             // when a skeleton is positioned with a head correctly, the basis coordinate system of this is:
00566             // x up, y left, z behind
00567             // Object headtracker (portable optitrack @ SSSA) has x left, y up, z forward (nose)
00568 
00569             // Compensation to object basis: (x left, y up, z front) VR:(x->z front, y->x left, z->y up)
00570             sceneData[ids[id]].azimuth = atan2(finaltrans.x,finaltrans.z)*180*M_1_PI; // overwrite azimuth and elevation with calculated values - Normal orientation
00571             sceneData[ids[id]].elevation = atan2(finaltrans.y,sqrt(finaltrans.z*finaltrans.z+finaltrans.x*finaltrans.x))*180*M_1_PI;
00572 #ifdef SCENEDBG
